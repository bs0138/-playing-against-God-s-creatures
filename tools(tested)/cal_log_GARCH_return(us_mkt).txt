# Load the 'rugarch' package
library(rugarch)

# Download S&P 500 data from Yahoo Finance
getSymbols("^GSPC", from = as.Date("2013-03-18"), to = Sys.Date())

# Extract the closing prices from the S&P 500 data
sp500 <- Cl(GSPC)

# Calculate the log returns of the S&P 500 prices
log_returns <- diff(log(sp500))

# Remove any missing data
log_returns <- na.omit(log_returns)

# Specify the GARCH model
spec <- ugarchspec(variance.model = list(model = "sGARCH", garchOrder = c(4, 2)), 
                   mean.model = list(armaOrder = c(0, 0)), 
                   distribution.model = "std")

# Fit the GARCH model to the log returns data
fit <- ugarchfit(spec, data = log_returns)

# Generate a forecast of the next 252 days of log returns
forecast <- ugarchforecast(fit, n.ahead = 252, data = log_returns)

# Extract the mean forecast returns from the forecast object
forecast_returns <- as.vector(forecast@forecast$seriesFor)

# Calculate the annualized mean forecast returns
mean_forecast_returns <- mean(forecast_returns) * 252

# Print the result
cat("The annualized mean forecast returns for the next 252 days are:", mean_forecast_returns, "\n")
